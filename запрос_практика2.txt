SELECT 
    з.НомерЗаказа,
    SUM(сз.Кол_во) AS Всего_товаров,
    SUM(сз.Сумма) AS Стоимость_товаров,
    м.Стоимость_материалов AS Стоимость_материалов,
    SUM(сз.Сумма) + м.Стоимость_материалов AS Полная_стоимость
FROM 
    Заказ з
    JOIN СоставЗаказа сз ON з.НомерЗаказа = сз.НомерЗаказа
    JOIN Товар т ON сз.Товар = т.UniqueID
    LEFT JOIN (
        SELECT 
            сз2.НомерЗаказа,
            SUM(сз2.Кол_во * п.Кол_во_материала * п.Себестоимость_мат) AS Стоимость_материалов
        FROM 
            СоставЗаказа сз2
            JOIN Товар т2 ON сз2.Товар = т2.UniqueID
            JOIN Производство п ON т2.Продукция = п.Продукция
        WHERE 
            сз2.НомерЗаказа = 1
        GROUP BY 
            сз2.НомерЗаказа
    ) м ON з.НомерЗаказа = м.НомерЗаказа
WHERE 
    з.НомерЗаказа = 1
GROUP BY 
    з.НомерЗаказа, м.Стоимость_материалов;
